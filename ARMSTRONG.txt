declare
n number:=&n;
sums number:=0;
rem number;
temp number:=n;
begin
while n>0 loop
rem:=mod(n,10);
sums:=sums+rem*rem*rem;
n:=trunc(n/10);
end loop;
if sums=temp then
dbms_output.put_line(temp||'is armstrong');
else
dbms_output.put_line(temp||'is not armstrong');
end if;
end;
